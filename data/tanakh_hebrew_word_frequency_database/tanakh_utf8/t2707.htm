האזנה לפרק זה

איוב פרק ז

א  הֲלֹא-צָבָא לֶאֱנוֹשׁ על- (עֲלֵי-) אָרֶץ;    וְכִימֵי שָׂכִיר יָמָיו.
ב  כְּעֶבֶד יִשְׁאַף-צֵל;    וּכְשָׂכִיר, יְקַוֶּה פָעֳלוֹ.
ג  כֵּן הָנְחַלְתִּי לִי, יַרְחֵי-שָׁוְא;    וְלֵילוֹת עָמָל, מִנּוּ-לִי.
ד  אִם-שָׁכַבְתִּי--    וְאָמַרְתִּי, מָתַי אָקוּם וּמִדַּד-עָרֶב;
וְשָׂבַעְתִּי נְדֻדִים    עֲדֵי-נָשֶׁף.
ה  לָבַשׁ בְּשָׂרִי רִמָּה, וגיש (וְגוּשׁ) עָפָר;    עוֹרִי רָגַע, וַיִּמָּאֵס.
ו  יָמַי קַלּוּ, מִנִּי-אָרֶג;    וַיִּכְלוּ, בְּאֶפֶס תִּקְוָה.
ז  זְכֹר, כִּי-רוּחַ חַיָּי;    לֹא-תָשׁוּב עֵינִי, לִרְאוֹת טוֹב.
ח  לֹא-תְשׁוּרֵנִי, עֵין רֹאִי;    עֵינֶיךָ בִּי וְאֵינֶנִּי.
ט  כָּלָה עָנָן, וַיֵּלַךְ;    כֵּן יוֹרֵד שְׁאוֹל, לֹא יַעֲלֶה.
י  לֹא-יָשׁוּב עוֹד לְבֵיתוֹ;    וְלֹא-יַכִּירֶנּוּ עוֹד מְקֹמוֹ.
יא  גַּם-אֲנִי,    לֹא אֶחֱשָׂךְ-פִּי:
אֲדַבְּרָה, בְּצַר רוּחִי;    אָשִׂיחָה, בְּמַר נַפְשִׁי.
יב  הֲיָם-אָנִי, אִם-תַּנִּין:    כִּי-תָשִׂים עָלַי מִשְׁמָר.
יג  כִּי-אָמַרְתִּי, תְּנַחֲמֵנִי עַרְשִׂי;    יִשָּׂא בְשִׂיחִי, מִשְׁכָּבִי.
יד  וְחִתַּתַּנִי בַחֲלֹמוֹת;    וּמֵחֶזְיֹנוֹת תְּבַעֲתַנִּי.
טו  וַתִּבְחַר מַחֲנָק נַפְשִׁי;    מָוֶת, מֵעַצְמוֹתָי.
טז  מָאַסְתִּי, לֹא-לְעֹלָם אֶחְיֶה;    חֲדַל מִמֶּנִּי, כִּי-הֶבֶל יָמָי.
יז  מָה-אֱנוֹשׁ, כִּי תְגַדְּלֶנּוּ;    וְכִי-תָשִׁית אֵלָיו לִבֶּךָ.
יח  וַתִּפְקְדֶנּוּ לִבְקָרִים;    לִרְגָעִים, תִּבְחָנֶנּוּ.
יט  כַּמָּה, לֹא-תִשְׁעֶה מִמֶּנִּי;    לֹא-תַרְפֵּנִי, עַד-בִּלְעִי רֻקִּי.
כ  חָטָאתִי, מָה אֶפְעַל לָךְ--    נֹצֵר הָאָדָם:
לָמָה שַׂמְתַּנִי לְמִפְגָּע לָךְ;    וָאֶהְיֶה עָלַי לְמַשָּׂא.
כא  וּמֶה, לֹא-תִשָּׂא פִשְׁעִי--    וְתַעֲבִיר אֶת-עֲ
